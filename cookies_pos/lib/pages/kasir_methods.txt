/* ADD THIS METHOD after _showProsesPesananPopup */
void _showPreOrderConfirmation() {
  if (_cart.isEmpty) {
    ScaffoldMessenger.of(context).showSnackBar(
      const SnackBar(content: Text('Keranjang masih kosong!')),
    );
    return;
  }
  _showProsesPesananPopup(isPreOrder: true);
}

/* MODIFY _prosesPesananSelesai to accept parameters */
Future<void> _prosesPesananSelesai(String paymentMethod, {bool isPreOrder = false}) async {
  final dbHelper = DatabaseHelper();
  final customerName = _customerNameController.text.trim();
  
  if (isPreOrder && customerName.isEmpty) {
    ScaffoldMessenger.of(context).showSnackBar(
      const SnackBar(content: Text('Nama pelanggan wajib diisi')),
    );
    return;
  }

  // Create sale items list
  final saleItems = <SaleItem>[];
  final List<String> itemDescriptions = [];
  final itemsJson = <Map<String, dynamic>>[];

  _cart.forEach((productId, qty) {
    final product = _products.firstWhere((p) => p.id == productId);
    itemDescriptions.add('${qty}x ${product.name}');
    
    saleItems.add(SaleItem(
      saleId: 0,
      productId: productId,
      productName: product.name,
      price: product.price,
      quantity: qty,
    ));
    
    itemsJson.add({
      'id': productId,
      'name': product.name,
      'price': product.price,
      'quantity': qty,
    });
  });

  if (isPreOrder) {
    // Save as PreOrder
    final preOrder = PreOrder(
      customerName: customerName.isEmpty ? 'Customer' : customerName,
      items: json.encode(itemsJson),
      totalAmount: _subtotal,
      createdAt: DateTime.now().toIso8601String(),
      status: 'pending',
    );
    
    await dbHelper.insertPreOrder(preOrder);
    
    setState(() {
      _cart.clear();
      _customerNameController.clear();
    });
    
    GlobalSync.instance.notify();
    
    if (!mounted) return;
    ScaffoldMessenger.of(context).showSnackBar(
      const SnackBar(
        content: Text('Pre-Order berhasil disimpan'),
        backgroundColor: Colors.green,
      ),
    );
  } else {
    // Regular sale - process immediately
    final sale = Sale(
      totalAmount: _subtotal,
      paymentMethod: paymentMethod,
      createdAt: DateTime.now().toIso8601String(),
    );

    await dbHelper.insertSale(sale, saleItems);

    final String detailedDesc = itemDescriptions.isNotEmpty
        ? 'Penjualan: ${itemDescriptions.join(", ")}'
        : 'Penjualan otomatis dari menu Kasir';

    await dbHelper.insertIncome(Income(
      amount: _subtotal,
      source: 'Penjualan POS',
      description: detailedDesc,
      date: DateTime.now().toIso8601String(),
    ));

    setState(() {
      _cart.clear();
    });

    await _loadProducts();
    GlobalSync.instance.notify();

    if (!mounted) return;
    ScaffoldMessenger.of(context).showSnackBar(
      const SnackBar(
        content: Text('Pesanan berhasil diproses'),
        backgroundColor: Colors.green,
      ),
    );
  }
}
